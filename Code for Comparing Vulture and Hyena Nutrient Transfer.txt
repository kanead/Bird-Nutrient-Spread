##------------------------------------------------------------------------##
## Foraging Radius for Hyenas
##------------------------------------------------------------------------##
e = 3.4*10^7 # energy density of meat J
Ema = 89 # BMR of adult Watts
Emc = 26 # BMR of juvenile 
k = 120 # cost of travel J/m
V=c(0.76, 2.03, 3.3) # walking speeds m/s

# Equation for foraging radius in meters is:
e*V/(3.3*(Ema+2*Emc)+2*k*V)
